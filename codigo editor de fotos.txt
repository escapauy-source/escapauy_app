    <script>
        // 1. Inicializar iconos de Lucide
        lucide.createIcons();

        // 2. Referencias a elementos del DOM
        const imageUpload = document.getElementById('imageUpload');
        const originalImage = document.getElementById('originalImage');
        const editedImage = document.getElementById('editedImage');
        const controls = document.getElementById('controls');
        const loader = document.getElementById('loader');
        const downloadLink = document.getElementById('downloadLink');
        const errorMessage = document.getElementById('errorMessage');

        // Nuevos elementos para Try-On
        const garmentUpload = document.getElementById('garmentUpload');
        const garmentPreview = document.getElementById('garmentPreview');
        const btnVirtualTryOn = document.getElementById('btn-virtual-try-on');

        // Botones de herramientas estándar
        const btnSquare = document.getElementById('btn-square');
        const btnVertical = document.getElementById('btn-vertical');
        const btnWhiteBg = document.getElementById('btn-white-bg');
        const btnSmile = document.getElementById('btn-smile');
        
        // NUEVO: Botón de reintentar
        const btnRetry = document.getElementById('btn-retry');

        // Variables de estado
        let base64ImageData = null; // Imagen de la persona
        let base64GarmentData = null; // Imagen de la prenda
        let lastApiCall = { type: null, prompt: null }; // Para el botón de reintentar

        // 3. Manejadores de carga de imagen

        // 3a. Imagen Principal (Persona)
        imageUpload.addEventListener('change', (event) => {
            const file = event.target.files[0];
            if (!file) return;

            const reader = new FileReader();
            reader.onload = (e) => {
                originalImage.src = e.target.result;
                base64ImageData = e.target.result.split(',')[1];
                controls.disabled = false;
                resetEditedImage();
            };
            reader.readAsDataURL(file);
        });
        
        // 3b. Imagen de Prenda (Garment)
        garmentUpload.addEventListener('change', (event) => {
            const file = event.target.files[0];
            if (!file) return;

            const reader = new FileReader();
            reader.onload = (e) => {
                garmentPreview.src = e.target.result;
                base64GarmentData = e.target.result.split(',')[1];
                errorMessage.textContent = '';
            };
            reader.readAsDataURL(file);
        });

        // 3c. Función para limpiar el estado de edición
        function resetEditedImage() {
            editedImage.src = "https://placehold.co/500x500/f3f4f6/a3a3a3?text=Lista+para+editar";
            downloadLink.classList.add('hidden');
            btnRetry.classList.add('hidden');
            errorMessage.textContent = '';
            lastApiCall = { type: null, prompt: null };
        }


        // 4. Función de reintento (Exponential Backoff) para la API
        async function fetchWithBackoff(url, options, maxRetries = 3) {
            let delay = 1000;
            for (let i = 0; i < maxRetries; i++) {
                try {
                    const response = await fetch(url, options);
                    if (response.ok) return await response.json();
                    if (response.status === 429 || response.status >= 500) {
                        console.warn(`Attempt ${i + 1} failed. Retrying in ${delay}ms...`);
                        await new Promise(resolve => setTimeout(resolve, delay));
                        delay *= 2;
                    } else {
                        throw new Error(`API Error: ${response.statusText} (Status ${response.status})`);
                    }
                } catch (error) {
                    if (i === maxRetries - 1) throw error;
                    console.warn(`Attempt ${i + 1} network error. Retrying in ${delay}ms...`);
                    await new Promise(resolve => setTimeout(resolve, delay));
                    delay *= 2;
                }
            }
        }
        
        // 5. El "Cerebro" de la IA (Nuestro prompt de sistema)
        const SYSTEM_PROMPT = `
# ROL Y OBJETIVO
Eres "E-com Stylist", un asistente de IA experto en edición fotográfica para moda y comercio electrónico. Tu objetivo es recibir una o más imágenes y una instrucción de texto. Tu tarea es generar UNA sola imagen de salida que aplique la instrucción de forma precisa, profesional y con calidad de catálogo.

# REGLAS DE CAPACIDAD Y EJECUCIÓN

## 1. Tarea: Expansión Cuadrada (Outpainting)
* **Instrucción:** "cuadrado 500x500"
* **Lógica:** Recibes UNA imagen. NO la recortes ni deformes. Expande la imagen (outpainting) rellenando el espacio faltante (lados o arriba/abajo) para crear un formato 1:1. El relleno debe ser coherente con el entorno.

## 2. Tarea: Expansión Vertical (Outpainting)
* **Instrucción:** "vertical instagram"
* **Lógica:** Igual que la Tarea 1, pero el objetivo es una relación de aspecto 4:5. Expande la imagen para rellenar ese formato.

## 3. Tarea: Fondo de Catálogo
* **Instrucción:** "fondo blanco"
* **Lógica:** Recibes UNA imagen. Realiza una segmentación de alta precisión del sujeto principal. Elimina el fondo original y reemplázalo con un blanco puro (#FFFFFF).

## 4. Tarea: Modificación de Expresión
* **Instrucción:** "hacer sonreír"
* **Lógica:** Recibes UNA imagen. Identifica el rostro y modifica sutilmente la expresión para crear una sonrisa natural, sin cambiar la identidad de la persona.

## 5. Tarea: Virtual Try-On (Poner Ropa)
* **Instrucción:** "virtual try-on" (O cualquier prompt que acompañe a dos imágenes)
* **Contexto:** Recibirás DOS imágenes: la Imagen 1 (Persona) y la Imagen 2 (Prenda).
* **Lógica:**
    1.  Esta es la tarea más importante. Analiza la prenda de la Imagen 2 (ej. una chaqueta en un fondo blanco, en una percha, o incluso en otro modelo).
    2.  Analiza la persona en la Imagen 1.
    3.  Tu tarea es "colocar" la prenda de la Imagen 2 sobre la persona de la Imagen 1.
    4.  Debes ajustar la prenda (forma, perspectiva, iluminación, arrugas) de forma fotorrealista al cuerpo y la pose de la persona en la Imagen 1.
    5.  Reemplaza la ropa que la persona lleva puesta por la nueva prenda.
    6.  El rostro de la persona, su peinado y el fondo de la Imagen 1 deben permanecer INTACTOS.

# FORMATO DE SALIDA
Genera únicamente la imagen editada. No añadas texto, bordes ni marcas de agua.
`;


        // 6. Función GENÉRICA para llamar a la API
        async function callApi(payload) {
            
            // IMPORTANTE: Pega tu clave API de Google AI Studio aquí
            // Ve a https://aistudio.google.com/ para obtener tu clave
            // ¡¡¡SÓLO NECESITAS PONERLA AQUÍ UNA VEZ!!!
            const apiKey = "TU_API_KEY_AQUI"; // <--- ¡REEMPLAZA ESTO!

            // Comprobación de seguridad para la clave API
            if (apiKey === "TU_API_KEY_AQUI") {
                errorMessage.textContent = "Error: No has configurado tu API Key en el código (busca 'TU_API_KEY_AQUI').";
                loader.style.display = 'none';
                return;
            }

            loader.style.display = 'flex';
            errorMessage.textContent = '';
            editedImage.src = "https://placehold.co/500x500/f3f4f6/a3a3a3?text=Procesando...";
            downloadLink.classList.add('hidden');
            btnRetry.classList.add('hidden');

            const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-image-preview:generateContent?key=${apiKey}`;

            try {
                const result = await fetchWithBackoff(apiUrl, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(payload)
                });

                // --- LÓGICA DE ERROR MEJORADA ---
                const candidate = result?.candidates?.[0];
                const base64Data = candidate?.content?.parts?.find(p => p.inlineData)?.inlineData?.data;

                if (base64Data) {
                    // ¡Éxito!
                    const imageUrl = `data:image/png;base64,${base64Data}`;
                    editedImage.src = imageUrl;
                    downloadLink.href = imageUrl;
                    downloadLink.classList.remove('hidden');
                    btnRetry.classList.remove('hidden'); // Mostrar botón de reintentar
                } else {
                    // --- Diagnóstico de Error Avanzado ---
                    console.error("Respuesta inesperada de la API:", result);
                    let errorMsg = "Error: No se pudo generar la imagen.";
                    
                    if (!candidate) {
                        // No hay 'candidate'. Podría ser un error 400 que devolvió JSON.
                        if (result.error) {
                            errorMsg += ` Detalles de Google: ${result.error.message}`;
                        } else {
                            errorMsg += " La API devolvió una respuesta vacía (sin 'candidates').";
                        }
                    } else if (candidate.finishReason && candidate.finishReason !== 'STOP' && candidate.finishReason !== 'MAX_TOKENS') {
                        // La IA se detuvo por una razón (ej. SAFETY)
                        errorMsg += ` Razón: La IA bloqueó el resultado (Motivo: ${candidate.finishReason}).`;
                        if (candidate.safetyRatings) {
                            errorMsg += " Revisa las calificaciones de seguridad en la consola.";
                        }
                    } else {
                        // Hay 'candidate' pero no hay 'inlineData'
                        errorMsg += " La API no devolvió datos de imagen ('inlineData') en la respuesta.";
                    }
                    
                    errorMessage.textContent = errorMsg;
                    // --- Fin del Diagnóstico ---
                }

            } catch (error) {
                console.error("Error llamando a la API de Gemini:", error);
                if (error.message.includes("403")) {
                     errorMessage.textContent = `Error 403 (Forbidden): Revisa que tu API Key sea correcta y que la API "Generative Language" esté habilitada en tu proyecto de Google Cloud.`;
                } else {
                    errorMessage.textContent = `Error al procesar la imagen: ${error.message}`;
                }
                editedImage.src = "https://placehold.co/500x500/f3f4f6/a3a3a3?text=Error";
            } finally {
                loader.style.display = 'none';
            }
        }

        // 7. Funciones de ayuda para construir Payloads
        
        // 7a. Tareas estándar (1 imagen)
        function standardTask(promptText) {
            if (!base64ImageData) {
                errorMessage.textContent = "Por favor, carga primero la imagen de la persona.";
                return;
            }
            // Guardar para "Reintentar"
            lastApiCall = { type: 'standard', prompt: promptText };
            
            const payload = {
                contents: [{
                    parts: [
                        { text: promptText },
                        { inlineData: { mimeType: "image/png", data: base64ImageData } }
                    ]
                }],
                generationConfig: { responseModalities: ['IMAGE'] },
                systemInstruction: { parts: [{ text: SYSTEM_PROMPT }] },
            };
            callApi(payload);
        }
        
        // 7b. Tarea Virtual Try-On (2 imágenes)
        function virtualTryOnTask() {
            if (!base64ImageData) {
                errorMessage.textContent = "Por favor, carga primero la imagen de la persona.";
                return;
            }
            if (!base64GarmentData) {
                errorMessage.textContent = "Por favor, carga la imagen de la prenda.";
                garmentUpload.click(); // Abrir el diálogo para cargarla
                return;
            }
            
            // Guardar para "Reintentar"
            lastApiCall = { type: 'try-on', prompt: null };

            const payload = {
                contents: [{
                    parts: [
                        { text: "virtual try-on: Coloca la prenda de la segunda imagen en la persona de la primera imagen." },
                        { inlineData: { mimeType: "image/png", data: base64ImageData } },    // Imagen 1 (Persona)
                        { inlineData: { mimeType: "image/png", data: base64GarmentData } } // Imagen 2 (Prenda)
                    ]
                }],
                generationConfig: { responseModalities: ['IMAGE'] },
                systemInstruction: { parts: [{ text: SYSTEM_PROMPT }] },
            };
            callApi(payload);
        }

        // 8. Asignar funciones a los botones
        
        // Botones con "data-prompt"
        [btnSquare, btnVertical, btnWhiteBg, btnSmile].forEach(button => {
            button.addEventListener('click', () => {
                const prompt = button.getAttribute('data-prompt');
                standardTask(prompt);
            });
        });

        // Botón de Virtual Try-On
        btnVirtualTryOn.addEventListener('click', () => {
            virtualTryOnTask();
        });
        
        // NUEVO: Botón de Reintentar
        btnRetry.addEventListener('click', () => {
            if (!lastApiCall.type) return;
            
            if (lastApiCall.type === 'standard') {
                standardTask(lastApiCall.prompt);
            } else if (lastApiCall.type === 'try-on') {
                virtualTryOnTask();
            }
        });

    </script>
